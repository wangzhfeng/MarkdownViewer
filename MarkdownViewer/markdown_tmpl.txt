// 阅读进度条功能
function updateReadingProgress() {
  var progressBar = document.getElementById('readingProgressBar');
  if (!progressBar) return;
  
  var scrollTop = window.pageYOffset || document.documentElement.scrollTop;
  var docHeight = document.documentElement.scrollHeight - document.documentElement.clientHeight;
  var scrollPercent = docHeight > 0 ? (scrollTop / docHeight) * 100 : 0;
  
  progressBar.style.width = scrollPercent + '%';
}

// 监听滚动更新进度
window.addEventListener('scroll', function() {
  updateReadingProgress();
});

  var help = document.getElementById('shortcutHelp');
  help.classList.add('visible');
  document.body.style.overflow = 'hidden';
}

function closeShortcutHelp(event) {
  if (event && event.target !== event.currentTarget) {
    return;
  }
  var help = document.getElementById('shortcutHelp');
  help.classList.remove('visible');
  document.body.style.overflow = '';
}

// Vim 风格快捷键支持
var vimMode = false;
var pendingKey = null;

// Vim 滚动功能
function vimScroll(key) {
  var scrollAmount = 0;
  
  switch(key) {
    case 'j': // 向下逐行
      window.scrollBy(0, 40);
      break;
    case 'k': // 向上逐行
      window.scrollBy(0, -40);
      break;
    case 'd': // 向下半屏
      window.scrollBy(0, window.innerHeight / 2);
      break;
    case 'u': // 向上半屏
      window.scrollBy(0, -window.innerHeight / 2);
      break;
    case 'f': // 向下整屏
      window.scrollBy(0, window.innerHeight);
      break;
    case 'b': // 向上整屏
      window.scrollBy(0, -window.innerHeight);
      break;
    case 'g': // 顶部（需要 gg）
      pendingKey = 'g';
      return;
    case 'G': // 底部
      window.scrollTo(0, document.body.scrollHeight);
      break;
    case 'h': // 向左
      window.scrollBy(-50, 0);
      break;
    case 'l': // 向右
      window.scrollBy(50, 0);
      break;
  }
  
  pendingKey = null;
}

// 大纲视图功能
var outlineVisible = false;
var outlineHeadings = [];
var activeHeadingIndex = -1;

// 生成大纲
function generateOutline() {
  var article = document.getElementById('markdownBody');
  var headings = article.querySelectorAll('h1, h2, h3, h4, h5, h6');
  var outlineTree = document.getElementById('outlineTree');
  
  outlineTree.innerHTML = '';
  outlineHeadings = [];
  
  headings.forEach(function(heading, index) {
    // 为没有 id 的标题生成 id
    if (!heading.id) {
      heading.id = 'heading-' + index;
    }
    
    outlineHeadings.push(heading);
    
    var li = document.createElement('li');
    var a = document.createElement('a');
    a.className = 'h' + heading.tagName.charAt(1);
    a.textContent = heading.textContent || heading.innerText;
    a.href = '#' + heading.id;
    a.onclick = function(e) {
      e.preventDefault();
      scrollToHeading(heading);
      highlightActiveHeading(index);
    };
    
    li.appendChild(a);
    outlineTree.appendChild(li);
  });
}

// 滚动到标题
function scrollToHeading(heading) {
  var offset = 80; // 顶部留白
  var elementPosition = heading.getBoundingClientRect().top + window.pageYOffset;
  window.scrollTo({
    top: elementPosition - offset,
    behavior: 'smooth'
  });
}

// 高亮当前激活的标题
function highlightActiveHeading(index) {
  var links = document.querySelectorAll('.outline-tree a');
  links.forEach(function(link, i) {
    link.classList.toggle('active', i === index);
  });
  activeHeadingIndex = index;
}

// 监听滚动，自动高亮大纲
function onScroll() {
  if (!outlineVisible || outlineHeadings.length === 0) return;
  
  var scrollPosition = window.pageYOffset + 100;
  
  for (var i = outlineHeadings.length - 1; i >= 0; i--) {
    var heading = outlineHeadings[i];
    if (heading.offsetTop <= scrollPosition) {
      if (activeHeadingIndex !== i) {
        highlightActiveHeading(i);
      }
      break;
    }
  }
}

// 切换大纲显示
function toggleOutline() {
  var panel = document.getElementById('outlinePanel');
  var body = document.getElementById('markdownBody');
  var toggleBtn = document.querySelector('.outline-toggle');
  
  outlineVisible = !outlineVisible;
  panel.classList.toggle('visible', outlineVisible);
  body.classList.toggle('with-outline', outlineVisible);
  toggleBtn.textContent = outlineVisible ? t('outlineClose') : t('outline');
  
  if (outlineVisible && outlineHeadings.length === 0) {
    generateOutline();
  }
  
  window.addEventListener('scroll', onScroll);
}

// 键盘快捷键支持
document.addEventListener('keydown', function(e) {
  // ? 键显示快捷键帮助
  if (e.key === '?' || e.keyCode === 191) {
    e.preventDefault();
    showShortcutHelp();
    return;
  }
  
  // 特殊处理 Shift+G（跳到底部）
  if (e.shiftKey && e.key === 'G') {
    e.preventDefault();
    vimScroll('G');
    return;
  }
  
  // 忽略其他带修饰键的组合
  if (e.ctrlKey || e.altKey || e.metaKey) {
    return;
  }
  
  // 忽略单独的 Shift 键
  if (e.shiftKey) {
    return;
  }
  
  // Vim 风格快捷键（小写字母）
  var vimKeys = ['j', 'k', 'd', 'u', 'f', 'b', 'g', 'h', 'l'];
  if (vimKeys.indexOf(e.key) !== -1) {
    e.preventDefault();
    
    // 处理 gg 组合（先按 g 再按 g）
    if (e.key === 'g') {
      if (pendingKey === 'g') {
        window.scrollTo(0, 0);
        pendingKey = null;
      } else {
        vimScroll('g');
      }
    } else {
      vimScroll(e.key);
    }
    return;
  }
  
  // Vim 风格快捷键（大写字母 G）
  if (e.key === 'G') {
    e.preventDefault();
    vimScroll('G');
    return;
  }
  
  // M 键切换布局
  if (e.keyCode === 77) {
    e.preventDefault();
    toggleLayout();
    return;
  }
  
  // T 键切换主题
  if (e.keyCode === 84) {
    e.preventDefault();
    toggleTheme();
    return;
  }
  
  // O 键切换大纲
  if (e.keyCode === 79) {
    e.preventDefault();
    toggleOutline();
    return;
  }
  
  // P 键导出 PDF
  if (e.keyCode === 80) {
    e.preventDefault();
    exportPdf();
    return;
  }
  
  // 数字键 1-6 快速跳转到对应级别的标题
  if (e.keyCode >= 49 && e.keyCode <= 54) {
    var level = e.keyCode - 48;
    var headings = document.querySelectorAll('h' + level);
    if (headings.length > 0) {
      e.preventDefault();
      scrollToHeading(headings[0]);
      // 找到对应的大纲项并高亮
      var index = outlineHeadings.indexOf(headings[0]);
      if (index >= 0) {
        highlightActiveHeading(index);
      }
    }
    return;
  }
  
  // ESC 键关闭图片查看器、快捷键帮助或预览
  if (e.keyCode === 27) {
    e.preventDefault();
    // 如果快捷键帮助打开，先关闭它
    var help = document.getElementById('shortcutHelp');
    if (help.classList.contains('visible')) {
      closeShortcutHelp({ target: help, currentTarget: help });
      return;
    }
    // 如果图片查看器打开，关闭它
    var viewer = document.getElementById('imageViewer');
    if (viewer.classList.contains('visible')) {
      closeImageViewer();
      return;
    }
    // 否则关闭预览
    window.chrome.webview.hostObjects.callback.OnKeyPressed(27);
    return;
  }
});

// Store the base directory for the original markdown file
var markdownBaseDir = "{0}";

// Support clicking on links to navigate
document.addEventListener('click', function(e) {
    var link = e.target.closest('a');
    if (!link) return;
    
    var href = link.getAttribute('href');
    if (!href) return;
    
    // Check if it's an external http/https link
    if (href.startsWith('http://') || href.startsWith('https://')) {
        e.preventDefault();
        
        // Notify host application to open in external browser
        window.chrome.webview.postMessage({
            type: 'externalLink',
            url: href
        });
        return;
    }
    
    // Check if it's a markdown file link
    if (href.match(/\.(md|markdown|mk)(\?.*)?$/i)) {
        e.preventDefault();
        
        // Resolve relative path using the original file's directory
        var resolvedPath;
        if (href.startsWith('file://')) {
            // Absolute file URL, use as-is
            resolvedPath = href;
        } else if (href.startsWith('/')) {
            // Absolute path
            resolvedPath = href.substring(1).replace(/\//g, '\\');
        } else {
            // Relative path - resolve against markdownBaseDir
            // Ensure baseDir ends with backslash
            var baseDir = markdownBaseDir;
            if (!baseDir.endsWith('\\')) {
                baseDir = baseDir + '\\';
            }
            
            // Normalize href: replace / with \ and resolve . and ..
            var normalizedHref = href.replace(/\//g, '\\');
            
            // Combine paths
            var combined = baseDir + normalizedHref;
            
            // Resolve .. by splitting and processing
            var parts = combined.split('\\');
            var result = [];
            
            // Handle drive letter (e.g., "d:")
            var driveLetter = null;
            if (parts.length > 0 && parts[0].match(/^[a-zA-Z]:$/)) {
                driveLetter = parts[0];
                parts.shift();
            }
            
            for (var i = 0; i < parts.length; i++) {
                var part = parts[i];
                if (part === '.' || part === '') {
                    continue;
                } else if (part === '..') {
                    if (result.length > 0) {
                        result.pop();
                    }
                } else {
                    result.push(part);
                }
            }
            
            resolvedPath = driveLetter ? driveLetter + '\\' + result.join('\\') : result.join('\\');
        }
        
        // Notify host application
        window.chrome.webview.postMessage({
            type: 'markdownLink',
            path: decodeURIComponent(resolvedPath)
        });
    }
});

// 显示 Toast 提示
function showToast(message, duration) {
  duration = duration || 3000;
  
  var toast = document.createElement('div');
  toast.className = 'toast-notification';
  toast.textContent = message;
  document.body.appendChild(toast);
  
  // 触发动画
  setTimeout(function() {
    toast.classList.add('visible');
  }, 10);
  
  // 自动消失
  setTimeout(function() {
    toast.classList.remove('visible');
    setTimeout(function() {
      document.body.removeChild(toast);
    }, 300);
  }, duration);
}

// Mermaid 流程图导出功能
function initMermaidExport() {
  var mermaidDivs = document.querySelectorAll('.mermaid');
  
  mermaidDivs.forEach(function(div, index) {
    // 避免重复添加
    if (div.querySelector('.mermaid-export-btn')) return;
    
    var btn = document.createElement('button');
    btn.className = 'mermaid-export-btn';
    btn.textContent = t('exportPng');
    btn.title = t('exportPng');
    btn.onclick = function(e) {
      e.preventDefault();
      e.stopPropagation();
      exportSingleMermaid(div);
    };
    
    div.appendChild(btn);
  });
}

// 导出单个流程图
function exportSingleMermaid(mermaidDiv) {
  // 创建临时容器
  var container = document.createElement('div');
  container.style.position = 'absolute';
  container.style.left = '-9999px';
  container.style.top = '-9999px';
  container.style.background = getComputedStyle(document.body).backgroundColor || '#ffffff';
  document.body.appendChild(container);
  
  // 获取 SVG
  var svg = mermaidDiv.querySelector('svg');
  
  if (!svg) {
    showToast(t('mermaidRendering'), 3000);
    document.body.removeChild(container);
    return;
  }
  
  // 克隆 SVG 并添加到临时容器
  var svgClone = svg.cloneNode(true);
  svgClone.style.maxWidth = 'none';
  container.appendChild(svgClone);
  
  // 使用 html2canvas 导出为图片
  showToast(t('exportingMermaid'), 5000);
  
  html2canvas(container, {
    backgroundColor: getComputedStyle(document.body).backgroundColor || '#ffffff',
    scale: 2,
    useCORS: true,
    allowTaint: true
  }).then(function(canvas) {
    // 转换为 PNG 数据 URL
    var pngDataUrl = canvas.toDataURL('image/png');
    
    // 创建下载链接
    var link = document.createElement('a');
    link.download = 'flowchart.png';
    link.href = pngDataUrl;
    link.click();
    
    document.body.removeChild(container);
    showToast(t('mermaidExportSuccess'), 3000);
  }).catch(function(error) {
    document.body.removeChild(container);
    showToast(t('mermaidExportFailed') + ': ' + error.message, 3000);
  });
}

// 代码复制功能
function initCodeCopy() {
  var codeBlocks = document.querySelectorAll('.markdown-body pre');
  
  codeBlocks.forEach(function(pre) {
    // 避免重复添加
    if (pre.querySelector('.code-copy-btn')) return;
    
    var btn = document.createElement('button');
    btn.className = 'code-copy-btn';
    btn.textContent = t('copy');
    btn.onclick = function(e) {
      e.preventDefault();
      e.stopPropagation();
      
      var code = pre.querySelector('code');
      if (!code) {
        code = pre;
      }
      
      var text = code.textContent || code.innerText;
      
      navigator.clipboard.writeText(text).then(function() {
        btn.textContent = t('copied');
        btn.classList.add('copied');
        
        setTimeout(function() {
          btn.textContent = t('copy');
          btn.classList.remove('copied');
        }, 2000);
      }).catch(function(err) {
        btn.textContent = t('copyFailed');
        setTimeout(function() {
          btn.textContent = t('copy');
        }, 2000);
      });
    };
    
    pre.appendChild(btn);
  });
}
