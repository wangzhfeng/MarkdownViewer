<script>
// 大纲视图功能
var outlineVisible = false;
var outlineHeadings = [];
var activeHeadingIndex = -1;

// 生成大纲
function generateOutline() {
  var article = document.getElementById('markdownBody');
  var headings = article.querySelectorAll('h1, h2, h3, h4, h5, h6');
  var outlineTree = document.getElementById('outlineTree');
  
  outlineTree.innerHTML = '';
  outlineHeadings = [];
  
  headings.forEach(function(heading, index) {
    // 为没有 id 的标题生成 id
    if (!heading.id) {
      heading.id = 'heading-' + index;
    }
    
    outlineHeadings.push(heading);
    
    var li = document.createElement('li');
    var a = document.createElement('a');
    a.className = 'h' + heading.tagName.charAt(1);
    a.textContent = heading.textContent || heading.innerText;
    a.href = '#' + heading.id;
    a.onclick = function(e) {
      e.preventDefault();
      scrollToHeading(heading);
      highlightActiveHeading(index);
    };
    
    li.appendChild(a);
    outlineTree.appendChild(li);
  });
}

// 滚动到标题
function scrollToHeading(heading) {
  var offset = 80; // 顶部留白
  var elementPosition = heading.getBoundingClientRect().top + window.pageYOffset;
  window.scrollTo({
    top: elementPosition - offset,
    behavior: 'smooth'
  });
}

// 高亮当前激活的标题
function highlightActiveHeading(index) {
  var links = document.querySelectorAll('.outline-tree a');
  links.forEach(function(link, i) {
    link.classList.toggle('active', i === index);
  });
  activeHeadingIndex = index;
}

// 监听滚动，自动高亮大纲
function onScroll() {
  if (!outlineVisible || outlineHeadings.length === 0) return;
  
  var scrollPosition = window.pageYOffset + 100;
  
  for (var i = outlineHeadings.length - 1; i >= 0; i--) {
    var heading = outlineHeadings[i];
    if (heading.offsetTop <= scrollPosition) {
      if (activeHeadingIndex !== i) {
        highlightActiveHeading(i);
      }
      break;
    }
  }
}

// 切换大纲显示
function toggleOutline() {
  var panel = document.getElementById('outlinePanel');
  var body = document.getElementById('markdownBody');
  var toggleBtn = document.querySelector('.outline-toggle');
  
  outlineVisible = !outlineVisible;
  panel.classList.toggle('visible', outlineVisible);
  body.classList.toggle('with-outline', outlineVisible);
  toggleBtn.textContent = outlineVisible ? '✕ 关闭' : '📑 大纲';
  
  if (outlineVisible && outlineHeadings.length === 0) {
    generateOutline();
  }
  
  window.addEventListener('scroll', onScroll);
}

// 键盘快捷键支持
document.addEventListener('keydown', function(e) {
  // O 键切换大纲
  if (e.keyCode === 79 && !e.ctrlKey && !e.altKey) {
    e.preventDefault();
    toggleOutline();
  }
  
  // 数字键 1-6 快速跳转到对应级别的标题
  if (e.keyCode >= 49 && e.keyCode <= 54 && !e.ctrlKey && !e.altKey) {
    var level = e.keyCode - 48;
    var headings = document.querySelectorAll('h' + level);
    if (headings.length > 0) {
      e.preventDefault();
      scrollToHeading(headings[0]);
      // 找到对应的大纲项并高亮
      var index = outlineHeadings.indexOf(headings[0]);
      if (index >= 0) {
        highlightActiveHeading(index);
      }
    }
  }
});

// Store the base directory for the original markdown file
var markdownBaseDir = "{0}";

// Support clicking on markdown file links to navigate
document.addEventListener('click', function(e) {
    var link = e.target.closest('a');
    if (!link) return;
    
    var href = link.getAttribute('href');
    if (!href) return;
    
    // Check if it's a markdown file link
    if (href.match(/\.(md|markdown|mk)(\?.*)?$/i)) {
        e.preventDefault();
        
        // Resolve relative path using the original file's directory
        var resolvedPath;
        if (href.startsWith('http://') || href.startsWith('https://') || href.startsWith('file://')) {
            // Absolute URL, use as-is
            resolvedPath = href;
        } else if (href.startsWith('/')) {
            // Absolute path
            resolvedPath = href.substring(1).replace(/\//g, '\\');
        } else {
            // Relative path - resolve against markdownBaseDir
            // Ensure baseDir ends with backslash
            var baseDir = markdownBaseDir;
            if (!baseDir.endsWith('\\')) {
                baseDir = baseDir + '\\';
            }
            
            // Normalize href: replace / with \ and resolve . and ..
            var normalizedHref = href.replace(/\//g, '\\');
            
            // Combine paths
            var combined = baseDir + normalizedHref;
            
            // Resolve .. by splitting and processing
            var parts = combined.split('\\');
            var result = [];
            
            // Handle drive letter (e.g., "d:")
            var driveLetter = null;
            if (parts.length > 0 && parts[0].match(/^[a-zA-Z]:$/)) {
                driveLetter = parts[0];
                parts.shift();
            }
            
            for (var i = 0; i < parts.length; i++) {
                var part = parts[i];
                if (part === '.' || part === '') {
                    continue;
                } else if (part === '..') {
                    if (result.length > 0) {
                        result.pop();
                    }
                } else {
                    result.push(part);
                }
            }
            
            resolvedPath = driveLetter ? driveLetter + '\\' + result.join('\\') : result.join('\\');
        }
        
        // Notify host application
        window.chrome.webview.postMessage({
            type: 'markdownLink',
            path: decodeURIComponent(resolvedPath)
        });
    }
});

// 页面加载完成后生成大纲
document.addEventListener('DOMContentLoaded', function() {
  generateOutline();
});
