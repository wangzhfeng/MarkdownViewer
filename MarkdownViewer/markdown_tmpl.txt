// 阅读进度条功能
function updateReadingProgress() {
  var progressBar = document.getElementById('readingProgressBar');
  if (!progressBar) return;
  
  var scrollTop = window.pageYOffset || document.documentElement.scrollTop;
  var docHeight = document.documentElement.scrollHeight - document.documentElement.clientHeight;
  var scrollPercent = docHeight > 0 ? (scrollTop / docHeight) * 100 : 0;
  
  progressBar.style.width = scrollPercent + '%';
}

// 监听滚动更新进度
window.addEventListener('scroll', function() {
  updateReadingProgress();
});

// 快捷键帮助功能
function showShortcutHelp() {
  var help = document.getElementById('shortcutHelp');
  help.classList.add('visible');
  document.body.style.overflow = 'hidden';
}

function closeShortcutHelp(event) {
  if (event && event.target !== event.currentTarget) {
    return;
  }
  var help = document.getElementById('shortcutHelp');
  help.classList.remove('visible');
  document.body.style.overflow = '';
}

// 图片放大查看功能
var currentImageSrc = '';

// 打开图片查看器
function openImageViewer(img) {
  var viewer = document.getElementById('imageViewer');
  var viewerImg = document.getElementById('imageViewerImg');
  var caption = document.getElementById('imageViewerCaption');
  
  currentImageSrc = img.src;
  viewerImg.src = currentImageSrc;
  
  // 如果有 alt 文本，显示为标题
  if (img.alt) {
    caption.textContent = img.alt;
    caption.style.display = 'block';
  } else {
    caption.style.display = 'none';
  }
  
  viewer.classList.add('visible');
  document.body.style.overflow = 'hidden'; // 禁止背景滚动
}

// 关闭图片查看器
function closeImageViewer() {
  var viewer = document.getElementById('imageViewer');
  viewer.classList.remove('visible');
  document.body.style.overflow = ''; // 恢复滚动
  currentImageSrc = '';
}

// 为所有图片添加点击事件

  var images = document.querySelectorAll('.markdown-body img');
  images.forEach(function(img) {
    img.onclick = function(e) {
      e.preventDefault();
      e.stopPropagation();

    };
  });
}

// Vim 风格快捷键支持
var vimMode = false;
var pendingKey = null;

// Vim 滚动功能
function vimScroll(key) {
  var scrollAmount = 0;
  
  switch(key) {
    case 'j': // 向下逐行
      window.scrollBy(0, 40);
      break;
    case 'k': // 向上逐行
      window.scrollBy(0, -40);
      break;
    case 'd': // 向下半屏
      window.scrollBy(0, window.innerHeight / 2);
      break;
    case 'u': // 向上半屏
      window.scrollBy(0, -window.innerHeight / 2);
      break;
    case 'f': // 向下整屏
      window.scrollBy(0, window.innerHeight);
      break;
    case 'b': // 向上整屏
      window.scrollBy(0, -window.innerHeight);
      break;
    case 'g': // 顶部（需要 gg）
      pendingKey = 'g';
      return;
    case 'G': // 底部
      window.scrollTo(0, document.body.scrollHeight);
      break;
    case 'h': // 向左
      window.scrollBy(-50, 0);
      break;
    case 'l': // 向右
      window.scrollBy(50, 0);
      break;
  }
  
  pendingKey = null;
}

// 布局模式功能
var currentLayout = 'centered';

// 切换布局
function toggleLayout() {
  var layoutToggle = document.querySelector('.layout-toggle');
  var body = document.getElementById('markdownBody');
  
  if (currentLayout === 'centered') {
    currentLayout = 'full';
    body.classList.remove('layout-centered');
    body.classList.add('layout-full');
    layoutToggle.textContent = '📐 宽屏';
  } else {
    currentLayout = 'centered';
    body.classList.remove('layout-full');
    body.classList.add('layout-centered');
    layoutToggle.textContent = '📐 居中';
  }
  
  // 保存布局偏好到 localStorage
  try {
    localStorage.setItem('markdownViewerLayout', currentLayout);
  } catch (e) {
    // Ignore localStorage errors
  }
}

// 加载保存的布局偏好
function loadLayout() {
  try {
    var savedLayout = localStorage.getItem('markdownViewerLayout');
    if (savedLayout === 'full') {
      currentLayout = 'full';
      var body = document.getElementById('markdownBody');
      body.classList.remove('layout-centered');
      body.classList.add('layout-full');
      var layoutToggle = document.querySelector('.layout-toggle');
      if (layoutToggle) {
        layoutToggle.textContent = '📐 宽屏';
      }
    }
  } catch (e) {
    // Ignore localStorage errors
  }
}

// 大纲视图功能
var outlineVisible = false;
var outlineHeadings = [];
var activeHeadingIndex = -1;

// 生成大纲
function generateOutline() {
  var article = document.getElementById('markdownBody');
  var headings = article.querySelectorAll('h1, h2, h3, h4, h5, h6');
  var outlineTree = document.getElementById('outlineTree');
  
  outlineTree.innerHTML = '';
  outlineHeadings = [];
  
  headings.forEach(function(heading, index) {
    // 为没有 id 的标题生成 id
    if (!heading.id) {
      heading.id = 'heading-' + index;
    }
    
    outlineHeadings.push(heading);
    
    var li = document.createElement('li');
    var a = document.createElement('a');
    a.className = 'h' + heading.tagName.charAt(1);
    a.textContent = heading.textContent || heading.innerText;
    a.href = '#' + heading.id;
    a.onclick = function(e) {
      e.preventDefault();
      scrollToHeading(heading);
      highlightActiveHeading(index);
    };
    
    li.appendChild(a);
    outlineTree.appendChild(li);
  });
}

// 滚动到标题
function scrollToHeading(heading) {
  var offset = 80; // 顶部留白
  var elementPosition = heading.getBoundingClientRect().top + window.pageYOffset;
  window.scrollTo({
    top: elementPosition - offset,
    behavior: 'smooth'
  });
}

// 高亮当前激活的标题
function highlightActiveHeading(index) {
  var links = document.querySelectorAll('.outline-tree a');
  links.forEach(function(link, i) {
    link.classList.toggle('active', i === index);
  });
  activeHeadingIndex = index;
}

// 监听滚动，自动高亮大纲
function onScroll() {
  if (!outlineVisible || outlineHeadings.length === 0) return;
  
  var scrollPosition = window.pageYOffset + 100;
  
  for (var i = outlineHeadings.length - 1; i >= 0; i--) {
    var heading = outlineHeadings[i];
    if (heading.offsetTop <= scrollPosition) {
      if (activeHeadingIndex !== i) {
        highlightActiveHeading(i);
      }
      break;
    }
  }
}

// 切换大纲显示
function toggleOutline() {
  var panel = document.getElementById('outlinePanel');
  var body = document.getElementById('markdownBody');
  var toggleBtn = document.querySelector('.outline-toggle');
  
  outlineVisible = !outlineVisible;
  panel.classList.toggle('visible', outlineVisible);
  body.classList.toggle('with-outline', outlineVisible);
  toggleBtn.textContent = outlineVisible ? '✕ 关闭' : '📑 大纲';
  
  if (outlineVisible && outlineHeadings.length === 0) {
    generateOutline();
  }
  
  window.addEventListener('scroll', onScroll);
}

// 键盘快捷键支持
document.addEventListener('keydown', function(e) {
  // ? 键显示快捷键帮助
  if (e.key === '?' || e.keyCode === 191) {
    e.preventDefault();
    showShortcutHelp();
    return;
  }
  
  // 特殊处理 Shift+G（跳到底部）
  if (e.shiftKey && e.key === 'G') {
    e.preventDefault();
    vimScroll('G');
    return;
  }
  
  // 忽略其他带修饰键的组合
  if (e.ctrlKey || e.altKey || e.metaKey) {
    return;
  }
  
  // 忽略单独的 Shift 键
  if (e.shiftKey) {
    return;
  }
  
  // Vim 风格快捷键（小写字母）
  var vimKeys = ['j', 'k', 'd', 'u', 'f', 'b', 'g', 'h', 'l'];
  if (vimKeys.indexOf(e.key) !== -1) {
    e.preventDefault();
    
    // 处理 gg 组合（先按 g 再按 g）
    if (e.key === 'g') {
      if (pendingKey === 'g') {
        window.scrollTo(0, 0);
        pendingKey = null;
      } else {
        vimScroll('g');
      }
    } else {
      vimScroll(e.key);
    }
    return;
  }
  
  // Vim 风格快捷键（大写字母 G）
  if (e.key === 'G') {
    e.preventDefault();
    vimScroll('G');
    return;
  }
  
  // M 键切换布局
  if (e.keyCode === 77) {
    e.preventDefault();
    toggleLayout();
    return;
  }
  
  // T 键切换主题
  if (e.keyCode === 84) {
    e.preventDefault();
    toggleTheme();
    return;
  }
  
  // O 键切换大纲
  if (e.keyCode === 79) {
    e.preventDefault();
    toggleOutline();
    return;
  }
  
  // P 键导出 PDF
  if (e.keyCode === 80) {
    e.preventDefault();
    exportPdf();
    return;
  }
  
  // 数字键 1-6 快速跳转到对应级别的标题
  if (e.keyCode >= 49 && e.keyCode <= 54) {
    var level = e.keyCode - 48;
    var headings = document.querySelectorAll('h' + level);
    if (headings.length > 0) {
      e.preventDefault();
      scrollToHeading(headings[0]);
      // 找到对应的大纲项并高亮
      var index = outlineHeadings.indexOf(headings[0]);
      if (index >= 0) {
        highlightActiveHeading(index);
      }
    }
    return;
  }
  
  // ESC 键关闭图片查看器、快捷键帮助或预览
  if (e.keyCode === 27) {
    e.preventDefault();
    // 如果快捷键帮助打开，先关闭它
    var help = document.getElementById('shortcutHelp');
    if (help.classList.contains('visible')) {
      closeShortcutHelp({ target: help, currentTarget: help });
      return;
    }
    // 如果图片查看器打开，关闭它
    var viewer = document.getElementById('imageViewer');
    if (viewer.classList.contains('visible')) {
      closeImageViewer();
      return;
    }
    // 否则关闭预览
    window.chrome.webview.hostObjects.callback.OnKeyPressed(27);
    return;
  }
});

// Store the base directory for the original markdown file
var markdownBaseDir = "{0}";

// Support clicking on links to navigate
document.addEventListener('click', function(e) {
    var link = e.target.closest('a');
    if (!link) return;
    
    var href = link.getAttribute('href');
    if (!href) return;
    
    // Check if it's an external http/https link
    if (href.startsWith('http://') || href.startsWith('https://')) {
        e.preventDefault();
        
        // Notify host application to open in external browser
        window.chrome.webview.postMessage({
            type: 'externalLink',
            url: href
        });
        return;
    }
    
    // Check if it's a markdown file link
    if (href.match(/\.(md|markdown|mk)(\?.*)?$/i)) {
        e.preventDefault();
        
        // Resolve relative path using the original file's directory
        var resolvedPath;
        if (href.startsWith('file://')) {
            // Absolute file URL, use as-is
            resolvedPath = href;
        } else if (href.startsWith('/')) {
            // Absolute path
            resolvedPath = href.substring(1).replace(/\//g, '\\');
        } else {
            // Relative path - resolve against markdownBaseDir
            // Ensure baseDir ends with backslash
            var baseDir = markdownBaseDir;
            if (!baseDir.endsWith('\\')) {
                baseDir = baseDir + '\\';
            }
            
            // Normalize href: replace / with \ and resolve . and ..
            var normalizedHref = href.replace(/\//g, '\\');
            
            // Combine paths
            var combined = baseDir + normalizedHref;
            
            // Resolve .. by splitting and processing
            var parts = combined.split('\\');
            var result = [];
            
            // Handle drive letter (e.g., "d:")
            var driveLetter = null;
            if (parts.length > 0 && parts[0].match(/^[a-zA-Z]:$/)) {
                driveLetter = parts[0];
                parts.shift();
            }
            
            for (var i = 0; i < parts.length; i++) {
                var part = parts[i];
                if (part === '.' || part === '') {
                    continue;
                } else if (part === '..') {
                    if (result.length > 0) {
                        result.pop();
                    }
                } else {
                    result.push(part);
                }
            }
            
            resolvedPath = driveLetter ? driveLetter + '\\' + result.join('\\') : result.join('\\');
        }
        
        // Notify host application
        window.chrome.webview.postMessage({
            type: 'markdownLink',
            path: decodeURIComponent(resolvedPath)
        });
    }
});
